\#26 Remove Duplicates from Sorted Array
```java
class Solution {
    public int removeDuplicates(int[] nums) {
        int l = 0;
        int r = 1;
        while (r < nums.length) {
            if (nums[l] == nums[r]) {
                r++;
            }
            else {
                l++;
                if (l != r) {
                    nums[l] = nums[r];
                }
                r++;
            }
        }
        return l + 1;
    }
}
```
