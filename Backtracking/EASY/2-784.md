\#784 Letter Case Permutation
```
class Solution {
    public List<String> letterCasePermutation(String S) {
        List<String> list = new ArrayList<>();
        int len = S.length();
        int letterC = 0;

        //1. count how many letters in S
        for (int i = 0; i < len; i++) {
            char c = S.charAt(i);
            if (Character.isDigit(c)) continue;
            letterC++;
        }

        //2. bit manipulation with total of 2^letterc (1 << letterc) possibility
        for (int j = 0; j < (1 << letterC); j++) {
            String newStr = "";
            int lc = 0; //iterate letter in S.
            for (int m = 0; m < len; m++) {
                char c = S.charAt(m);
                if (Character.isDigit(c)) { //mth character in S is digit
                    newStr = newStr + c;
                }
                else { //mth character in S is letter
                    if ((j & (1 << lc)) == 0) { //`lc`th is 0, lowercase
                        c = Character.toLowerCase(c);
                        newStr = newStr + c;
                    }
                    else { //`lc`th is 1, uppercase
                        c = Character.toUpperCase(c);
                        newStr = newStr + c;
                    }
                    lc++;
                }
            }
            list.add(newStr);
        }
        return list;
    }
}
```
