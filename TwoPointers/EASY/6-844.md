\#844 Backspace String Compare
1. two pointers
```
class Solution {
    public boolean backspaceCompare(String S, String T) {
        int s = S.length() - 1;
        int t = T.length() - 1;
        int sc = 0;
        int tc = 0;
        while (s >= 0 || t >= 0) {
            while (s >= 0) {
                if (S.charAt(s) == '#') {
                    sc++;
                    s--;
                }
                else if (sc > 0) {
                    s--;
                    sc--;
                }
                else break;
            }
            while (t >= 0) {
                if (T.charAt(t) == '#') {
                    tc++;
                    t--;
                }
                else if (tc > 0) {
                    t--;
                    tc--;
                }
                else break;
            }
            if (s >= 0 && t >= 0 && S.charAt(s) != T.charAt(t)) return false;
            if ((s < 0 && t >= 0) || (s >= 0 && t < 0)) return false;
            s--;
            t--;
        }
        return true;
    }
}
```
